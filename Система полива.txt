bool button = 0; // Объявляем переменную состояния кнопки

void setup() {
  pinMode(3, OUTPUT); // 3 пин на вывод
  pinMode(13, OUTPUT); // 13 пин на вывод (встроенный светодиод)
  digitalWrite(13, 1); // Включаем светодиод
  pinMode(4, INPUT); // 4 пин на вход
}

void loop() {
  if (digitalRead(4) == 1) { // Если на 4 пин приходит 5В (кнопка нажата)
    button =1; // Меняем состояние кнопки
  };
  if (button == 1){ // Если состояние кнопки нажата
    digitalWrite(3, 1); // Посылаем управляющий сигнал на затвор мосфета
    digitalWrite(13, 0); // Выключаем светодиод
    delay(10000); // Ждём 10 секунд
    digitalWrite(3, 0); // Вырубаем управляющий сигнал
    digitalWrite(13, 1); // Врубаем светодиод
    button = 0; // Обнуляем состояние кнопки
  };
}
